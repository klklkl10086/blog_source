## 快速排序

```c++
void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int i = l - 1, j = r + 1, x = q[l + r >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j), quick_sort(q, j + 1, r);
    //用j的情况下应当保证x取不到左边界，用i要保证取不到右边界，
}
```

## 浮点数二分

控制精度来控制循环，精度要比题目所要求的多2（y总经验）

```c++
bool check(double x) {/* ... */} // 检查x是否满足某种性质

double bsearch_3(double l, double r)
{
    const double eps = 1e-6;   // eps 表示精度，取决于题目对精度的要求
    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        if (check(mid)) r = mid;
        else l = mid;
    }
    return l;
}
```

```c++
 ll quick_power(ll n, ll p){
    ll res = 1;
    while(p){  //如果指数不为0
        if(p & 1) res *= n;  // 如果对应二进制数的当前位为1, 则res 乘以当前位的值

        p >>= 1;             //指数右移1位
```

```cpp
void merge(vector<PII> &segs)
{
    vector<PII> res;

    sort(segs.begin(), segs.end());

    int st = -2e9, ed = -2e9;//最左边
    for (auto seg : segs)
        if (ed < seg.first)
        {
            if (st != -2e9) res.push_back({st, ed});
            st = seg.first, ed = seg.second;
        }
        else ed = max(ed, seg.second);

    if (st != -2e9) res.push_back({st, ed});

    segs = res;
}
```

### 双链表

```cpp
// e[]表示节点的值，l[]表示节点的左指针，r[]表示节点的右指针，idx表示当前用到了哪个节点
int e[N], l[N], r[N], idx;

// 初始化
void init()
{
    //0是左端点，1是右端点
    r[0] = 1, l[1] = 0;
    idx = 2;
}

// 在节点a的右边插入一个数x
void insert(int a, int x)
{
    e[idx] = x;
    l[idx] = a, r[idx] = r[a];
    l[r[a]] = idx, r[a] = idx ++ ;
}

// 删除节点a
void remove(int a)
{
    l[r[a]] = l[a];
    r[l[a]] = r[a];
}
```

```cpp
// h[N]存储堆中的值, h[1]是堆顶，x的左儿子是2x, 右儿子是2x + 1
// ph[k]存储第k个插入的点在堆中的位置  数组下标-->堆中的下标

int h[N], ph[N], hp[N], size;

// 交换两个点，及其映射关系
void heap_swap(int a, int b)
{
    swap(ph[hp[a]],ph[hp[b]]);
    swap(hp[a], hp[b]);
    swap(h[a], h[b]);
}
//小根堆
void down(int u)
{
    int t = u;
    if (u * 2 <= size && h[u * 2] < h[t]) t = u * 2;//与左边子节点进行比较取小的
    if (u * 2 + 1 <= size && h[u * 2 + 1] < h[t]) t = u * 2 + 1;//与右边子节点比较取小的
    if (u != t)//将小的变为父节点
    {
        heap_swap(u, t);
        down(t);
    }
}

void up(int u)
{
    while (u / 2 && h[u] < h[u / 2])
    {
        heap_swap(u, u / 2);
        u >>= 1;
    }
}
```

# 树状数组

在线,支持修改（前缀和不支持修改）,使用的时候**下标要从1开始**



#### 用处



在O(logn)的时间内:

1. 给指定位置上的数字加上一个数字(**单点修改**)
2. 求指定的前缀和(**区间查询**)



可以通过转化例如利用差分转换为单点查询区间修改等。



对于树状数组,每个节点i的父节点是$i+lowbit(i)$

#### 计算

$tr[x]=\sum_{x-lowbit(x)+1}^{x}{a[i]}$

$lowbit(x)=x\&(-x)$

#### 基本操作

```cpp
int lowbit(int x)
{
    return x& -x;
}
void add(int x,int v)
{
    for(int i=x;i<=n;i+=lowbit(i)) tr[i]+=v;
}
int query(int x)//返回[1,x]的和
{
    int res=0;
    for(int i=x;i>0;i-=lowbit(i))res+=tr[i];
    return res;
}
```

### 操作

在**O(logn)**时间内完成:

1. 单点修改

```cpp
void pushup(int u)//更新一下当前节点的sum,当前节点的编号
{
	tr[u].sum=tr[u*2].sum+tr[u*2+1].sum;
}
void build(int u,int l,int r)//当前节点,当前节点对应的范围
{
	if(l==r) tr[u]={l,r,w[l]};
	else
	{
		int mid = (l+r)/2;
		tr[u]={l,r};
		build(u*2,l,mid),build(u*2+1,mid+1,r);
		pushup(u);
	}
}
void modify(int u,int x,int v)//当前节点,要修改的节点,要加的值
{
	if(tr[u].l==tr[u].r) tr[u].sum+=v;//找到了要修改的节点
	else
	{
		int mid = tr[u].l+tr[u].r>>1;
		if(x<=mid)modify(u*2,x,v);
		else modify(u*2+1,x,v);
		pushup(u);
	}
}
int query(int u,int l,int r) //当前节点,要查的区间
{
	if(tr[u].l>=l&&tr[u].r<=r) return tr[u].sum;
	int mid = tr[u].l+tr[u].r>>1;
	int sum=0;
	if(l<=mid) sum=query(u*2,l,r);
	if(r>=mid+1) sum+=query(u*2+1,l,r);
	return sum;
}
```
